#include <iostream>
#include <iomanip>
#define _USE_MATH_DEFINES
#include <math.h>
using namespace std;
int main()
{//Y=tg x3 + 2sinx - 3
	double x, y, x1, x2, h;
	cout << "Enter x1, x2, h:" << endl;
	cin >> x1 >> x2 >> h;
	x = x1;
	while (x <= x2) {
		y = tan(pow(x, 3)) + 2 * sin(x) - 3;
		cout << setw(10) << x << setw(15) << y << endl;
		x += h;
	}
	do {
		y = tan(pow(x, 3)) + 2 * sin(x) - 3;
		cout << setw(10) << x << setw(15) << y << endl;
		x += h;
	} while (x <= x2 + h / 2);

}
