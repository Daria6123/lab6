#include <iostream>
#include <windows.h>
using namespace std;
int main()
{
    SetConsoleOutputCP(1251);
    double a, b, c, x, f;
    cout << "Enter a, b, c, x:";
    cin >> a >> b >> c >> x;
    if (x < 6 && b + c != 0 && c!=0) {
        f = ((a * pow(x, 3)) / c) + (b);
        cout << "F =" << f;
        if (x>6 && (b + c) == 0 && b!=0) {
            f = (a * pow(x, 2)) - (c) / (b);
            cout << "F =" << f;
        }
    }
    else {
        f = (x) / (-c + 2);
        cout << "F =" << f;
    }
}
